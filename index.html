let currentStep = 0;

const totalSteps = steps.length;


function renderProgressTrack(){

  const track = document.getElementById('progressTrack');

  let html = '';

  steps.forEach((s,i)=>{

    const cls = i < currentStep ? 'done' : i === currentStep ? 'active' : '';

    const humanCls = s.isHuman ? 'human-step' : '';

    html += `<div class="progress-step ${cls} ${humanCls}" onclick="jumpTo(${i})"><div class="progress-dot"></div><span style="display:none">${s.num}</span></div>`;

    if(i < steps.length -1) html += `<div class="progress-line"></div>`;

  });

  track.innerHTML = html;

}


function renderScreen(){

  const s = steps[currentStep];

  const pane = document.getElementById('screenPane');

  pane.scrollTop = 0;

  pane.innerHTML = `

    <div class="step-header">

      <div class="step-label">

        <span class="step-num ${s.isHuman?'human':''}">Step ${s.num} of ${totalSteps}</span>

        <span class="role-badge ${s.roleClass}">${s.role}</span>

        ${s.isHuman ? '<span class="badge" style="background:var(--human);color:#fff;font-size:9px">HUMAN-IN-THE-LOOP</span>' : ''}

      </div>

      <div class="step-title">${s.title}</div>

      <div class="step-desc">${s.desc}</div>

    </div>

    ${s.screen}

  `;

}


function renderAgents(){

  const s = steps[currentStep];

  const body = document.getElementById('agentBody');

  body.scrollTop = 0;

  const autoPlay = document.getElementById('autoPlay').checked;


  let html = '';

  s.agents.forEach((a,i)=>{

    const statusLabel = a.status === 'active' ? (a.type==='human'?'Human action':'Processing') : a.status === 'pending' ? 'Queued' : 'Done';

    html += `<div class="agent-event ${a.type} ${a.status}" id="agent-${i}" style="transition-delay:${autoPlay ? (i*0.6)+'s' : '0s'}">`;

    html += `<div class="agent-label">${a.label || statusLabel}</div>`;

    html += `<div class="agent-name">${a.name}</div>`;

    if(a.reasoning){

      html += `<div class="agent-reasoning">"${a.reasoning}"</div>`;

    }

    if(a.outputs){

      html += `<ul class="agent-outputs">${a.outputs.map(o=>`<li>${o}</li>`).join('')}</ul>`;

    }

    if(a.flag){

      html += `<div class="agent-flag ${a.flag.type}">${a.flag.text}</div>`;

    }

    if(a.meta){

      html += `<div class="agent-meta">${a.meta}</div>`;

    }

    html += `</div>`;

  });

  body.innerHTML = html;


  // Animate in

  setTimeout(()=>{

    s.agents.forEach((a,i)=>{

      const el = document.getElementById('agent-'+i);

      if(el){

        const delay = autoPlay ? i * 600 : i * 150;

        setTimeout(()=> el.classList.add('visible'), delay);

      }

    });

  }, 100);

}


function updateNav(){

  document.getElementById('stepCurrent').textContent = currentStep + 1;

  document.getElementById('stepTotal').textContent = totalSteps;

  document.getElementById('prevBtn').disabled = currentStep === 0;

  document.getElementById('prevBtn').style.opacity = currentStep === 0 ? '.4' : '1';

  document.getElementById('nextBtn').textContent = currentStep === totalSteps -1 ? 'Journey complete ✓' : 'Next step →';

  document.getElementById('nextBtn').disabled = currentStep === totalSteps -1;

  document.getElementById('nextBtn').style.opacity = currentStep === totalSteps -1 ? '.4' : '1';

}


function go(dir){

  const next = currentStep + dir;

  if(next < 0 || next >= totalSteps) return;

  currentStep = next;

  render();

}

function jumpTo(i){

  currentStep = i;

  render();

}

function render(){

  renderProgressTrack();

  renderScreen();

  renderAgents();

  updateNav();

}

  if(e.key === 'ArrowRight' || e.key === ' ') { e.preventDefault(); go(1); }

  if(e.key === 'ArrowLeft') { e.preventDefault(); go(-1); }

});
